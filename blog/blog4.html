Step 2 — Export YOLOv8 to TensorRT

export_tensorrt.py — Convert to TensorRT Engine
from ultralytics import YOLO

# Load YOLOv8 nano model
model = YOLO("yolov8n.pt")

# Export to TensorRT with INT8 quantization
model.export(
    format="engine",
    device=0,           # GPU 0 (Orin Ampere)
    half=True,          # FP16 precision
    imgsz=640,          # Input image size
    workspace=4,        # TensorRT workspace GB
    verbose=True
)
print("TensorRT engine saved: yolov8n.engine")

Step 3 — Real-Time Detection with Camera

detect_camera.py — Live Object Detection
from ultralytics import YOLO
import cv2

# Load TensorRT engine
model = YOLO("yolov8n.engine")

# Open USB camera
cap = cv2.VideoCapture(0)
cap.set(cv2.CAP_PROP_FRAME_WIDTH,  640)
cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)

while True:
    ret, frame = cap.read()
    if not ret: break

    # Run YOLOv8 inference
    results = model.predict(frame, conf=0.5,
                              device=0, verbose=False)

    # Draw bounding boxes
    annotated = results[0].plot()
    cv2.imshow("YOLOv8 — Jetson Orin", annotated)

    for box in results[0].boxes:
        cls  = results[0].names[int(box.cls)]
        conf = float(box.conf)
        print(f"Detected: {cls} ({conf:.2f})")

    if cv2.waitKey(1) == ord('q'): break

cap.release()
cv2.destroyAllWindows()

Step 4 — Publish Detections as ROS2 Topics

yolo_ros2_node.py — ROS2 Detection Publisher
import rclpy
from rclpy.node import Node
from std_msgs.msg import String
from ultralytics import YOLO
import cv2, json

class YoloNode(Node):
    def __init__(self):
        super().__init__("yolo_detector")
        self.model = YOLO("yolov8n.engine")
        self.pub   = self.create_publisher(
                        String, "/detections", 10)
        self.cap   = cv2.VideoCapture(0)
        self.create_timer(0.033, self.detect) # 30 Hz

    def detect(self):
        ret, frame = self.cap.read()
        if not ret: return
        results = self.model.predict(
                    frame, conf=0.5, device=0,
                    verbose=False)
        detections = []
        for box in results[0].boxes:
            detections.append({
                "class": results[0].names[int(box.cls)],
                "conf":  round(float(box.conf), 2),
                "bbox":  box.xyxy[0].tolist()
            })
        msg = String()
        msg.data = json.dumps(detections)
        self.pub.publish(msg)

def main():
    rclpy.init()
    rclpy.spin(YoloNode())

if __name__ == "__main__":
    main()
